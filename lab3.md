## 7. Ход работы:

```
ping 11.23.89.213
ssh test@11.23.89.213
cd code1
ls -alhF
mkdir test
cp *.js* test/
pax -w -x tar  test > arch.tar
rm -rf test/
exit
scp test@11.23.89.213:/home/test/code1/arch.tar .
tar -tvf arch.tar
tar -xf arch.tar
cd test/
rm settings.js
mv main.js test_main.js
cd ..
tar -czf test.tar.gz test/
sftp test@11.23.89.213
cd code1
put test.tar.gz
exit
ssh test@11.23.89.213
cd code1
tar -xzf test.tar.gz
ls -alhF test/
exit
exit
```
